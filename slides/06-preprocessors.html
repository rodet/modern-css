5. CSS Preprocessors ⚙️

LESS, SASS, PostCSS
Pre-processors brought modularity forward with nesting and reusable blocks (mixins, functions, etc)

Nesting

// Sass
.class {
  color: #bada55;
  a {
    color: #c0ffee;
    &:hover {
      color: #ac1d1c;
    }
  }
  &--connected {
    color: #0ff1ce;
  }
}
// Compiled CSS
.class {
  color: #bada55;
}
.class a {
  color: #c0ffee;
}
.class a:hover {
  color: #ac1d1c;
}
.class--connected {
  color: #0ff1ce;
}

Imports

// Sass, _foo.scss
.foo {
  content: 'Hello World';
}
// Sass, bar/_baz.scss
.baz {
  content: 'Goodbye World';
}

// Sass, style.scss
@import 'foo';

.waldo {
  content: 'Where is Waldo?'
}

@import 'bar/baz';
// CSS, style.css
.foo {
  content: 'Hello World';
}

.waldo {
  content: 'Where is Waldo?'
}

.baz {
  content: 'Goodbye World';
}

Reusability

Functions

// Sass
@function rwd($target, $context) {
  @return $target / $context * 100%;
}

.sidebar {
  width: rwd(300, 940);
}

Mixins

// Sass
@mixin button($color) {
  $border: mix(black, $color, 40%);

  background: $color;
  border: 2px solid $border;

  &:hover {
    background: mix(white, $color, 25%);
    border-color: mix(white, $border, 25%);
  }
}

.button--error {
  @include button(red);
}
